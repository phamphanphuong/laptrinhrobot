Giải Pháp IoT Phòng Khám (Smart Clinic)

Điểm khác biệt của MediTech là khả năng lập trình vi điều khiển (ESP32 / Raspberry Pi Pico) để đọc tín hiệu trực tiếp từ thiết bị y tế (SpO2, Nhịp tim), sau đó truyền dữ liệu lên máy chủ Web Backend (Spring Boot/Java). Bác sĩ có thể theo dõi biểu đồ Real-time của toàn bộ bệnh nhân thông qua màn hình Dashboard.

  • Truyền thông MQTT / WebSockets siêu tốc.
  • Xử lý Data nội bộ khép kín, bảo mật dữ liệu y tế.
  • Cảnh báo qua Telegram/Zalo khi sinh hiệu bất thường.
ESP32_Medical_Firmware.cpp 
#include <WiFi.h>
#include <PubSubClient.h>
#include "MAX30105.h" // Thư viện cảm biến

void loop() {
  int heartRate = particleSensor.getHeartRate();
  int spo2 = particleSensor.getSpO2();

  if (spo2 < 90) {
    triggerLocalAlarm(); // Kích hoạt còi tại chỗ
  }

  // Đẩy dữ liệu lên Spring Boot Server
  String payload = "{\"id\":\"B12\", \"hr\":" + 
                    String(heartRate) + ", \"spo2\":" + 
                    String(spo2) + "}";
  mqttClient.publish("hospital/icu/vitals", payload.c_str());
  
  delay(2000);
}
